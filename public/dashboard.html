const express = require("express"); const sqlite3 = require("sqlite3").verbose(); const bodyParser = require("body-parser"); const cors = require("cors"); const path = require("path"); const app = express(); const PORT = 3000; app.use(cors()); app.use(bodyParser.json()); app.use(express.static("public")); const db = new sqlite3.Database("./database.db"); db.serialize(() => { db.run(` CREATE TABLE IF NOT EXISTS equipos( id INTEGER PRIMARY KEY AUTOINCREMENT, nombre TEXT, serial TEXT, area TEXT, descripcion TEXT, estado TEXT ) `); db.run(` CREATE TABLE IF NOT EXISTS empleados( id INTEGER PRIMARY KEY AUTOINCREMENT, nombre TEXT, cargo TEXT, area TEXT ) `); db.run(` CREATE TABLE IF NOT EXISTS asignaciones( id INTEGER PRIMARY KEY AUTOINCREMENT, equipo_id INTEGER, empleado_id INTEGER, fecha TEXT ) `); db.run(` CREATE TABLE IF NOT EXISTS mantenimientos( id INTEGER PRIMARY KEY AUTOINCREMENT, equipo_id INTEGER, tipo TEXT, descripcion TEXT, fecha TEXT ) `); }); // EQUIPOS app.get("/equipos",(req,res)=>{ db.all("SELECT * FROM equipos",(err,rows)=>{ res.json(rows); }); }); app.post("/equipos",(req,res)=>{ const {nombre,serial,area,descripcion}=req.body; db.run( `INSERT INTO equipos(nombre,serial,area,descripcion,estado) VALUES(?,?,?,?,?)`, [nombre,serial,area,descripcion,"Activo"] ); res.json({mensaje:"Equipo creado"}); }); app.delete("/equipos/:id",(req,res)=>{ db.run("DELETE FROM equipos WHERE id=?",[req.params.id]); res.json({mensaje:"Equipo eliminado"}); }); // MANTENIMIENTO app.post("/mantenimiento",(req,res)=>{ const {equipo_id,tipo,descripcion,fecha}=req.body; db.run( `INSERT INTO mantenimientos(equipo_id,tipo,descripcion,fecha) VALUES(?,?,?,?)`, [equipo_id,tipo,descripcion,fecha] ); res.json({mensaje:"Mantenimiento registrado"}); }); app.get("/mantenimiento",(req,res)=>{ db.all("SELECT * FROM mantenimientos",(err,rows)=>{ res.json(rows); }); }); // ASIGNACION app.post("/asignar",(req,res)=>{ const {equipo_id,empleado_id,fecha}=req.body; db.run( `INSERT INTO asignaciones(equipo_id,empleado_id,fecha) VALUES(?,?,?)`, [equipo_id,empleado_id,fecha] ); res.json({mensaje:"Equipo asignado"}); }); // SERVIDOR app.listen(PORT,()=>{ console.log("Servidor activo en puerto "+PORT); });
